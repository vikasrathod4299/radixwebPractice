2. Calculate area of circle.

fun calculateAreaOfCircle(ind radius):
{
    PI = 3.14
    r = radius

    float area = PI * r * r
    float circumferences = 2 * PI * r

    return area, circumferences
}

4. Sum of even numbers 

fun sumOfEvenNumbers(Array nums)
{
    sum=0
    for (int i = 0; i<nums.length; i++ ):
        if nums[i]%2==0
        {
            sum = sum + nums[i]
        }
    
    return sum
}
